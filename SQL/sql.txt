24. Display the department numbers of departments which have more than one clerk.
----SELECT DEPTNO
   	FROM EMP
   	WHERE EXISTS (SELECT COUNT(JOB) AS CLERK_COUNT 
        FROM EMP
        WHERE JOB = 'CLERK'
        GROUP BY DEPTNO)
   	GROUP BY DEPTNO;
